-- ══════════════════════════════════════════
--              SCRIPTS TAB
-- ══════════════════════════════════════════
local ScriptsTab = Window:CreateTab("Scripts")
local ExecSub    = ScriptsTab:CreateSubTab("Execute")
local ExecSection = ExecSub:CreateSector("Execute")

ExecSection:AddTextbox("Script URL", nil, function(v) end, "ScriptURL")
ExecSection:AddButton("Load Script", function()
    local url = Library.flags["ScriptURL"]
    if url and url ~= "" then
        local ok, err = pcall(function() loadstring(game:HttpGet(url))() end)
        if not ok then Window:Notify("Erreur", "Impossible de charger.", 4) end
    else
        Window:Notify("Erreur", "Entre une URL valide.", 4)
    end
end)

-- ══════════════════════════════════════════
Window:Notify("Photon", "Script chargé avec succès.", 5)
